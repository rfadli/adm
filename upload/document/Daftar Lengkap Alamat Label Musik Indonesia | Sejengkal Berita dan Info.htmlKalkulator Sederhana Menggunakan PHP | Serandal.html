<?php

$nilai1 = $_POST['nilai1'];
$nilai2 = $_POST['nilai2'];
$operator = $_POST['operator'];

if ($operator == "Penjumlahan")
{
 $hasil = $nilai1 + $nilai2;
 $tanda = "+"; 
 }
else if ($operator == "Pengurangan")
{
 $hasil = $nilai1 - $nilai2;
 $tanda = "-";
 }
else if ($operator == "Perkalian")
{
 $hasil = $nilai1 * $nilai2;
 $tanda = "*"; 
 }
else if ($operator == "Pembagian")
{
 $hasil = $nilai1 / $nilai2;
 $tanda = "/";
 }


 ?>
